## 🔥 المرحلة 1: Firebase Architecture

### **المهمة 1.1: تصميم قاعدة البيانات**

**الغرض:**  
قبل كتابة أي كود، نصمم هيكل البيانات بالكامل.

**ما هو Firestore؟**  
قاعدة بيانات NoSQL تعمل بنظام **Collections** و **Documents**.  
- Collection = جدول (لكن مرن)  
- Document = صف (JSON)  
- يدعم Subcollections (مجموعات داخل مجموعات)

**هيكل البيانات المطلوب:**

```
firestore/
│
├── users/                        # المستخدمون
│   └── {userId}/                 # Document لكل مستخدم
│       ├── id: string
│       ├── name: string
│       ├── email: string
│       ├── role: 'student' | 'admin' | 'batch_leader'
│       ├── university: string
│       ├── faculty: string
│       ├── year: number
│       ├── batch: string          # الدفعة (مثل: '2024')
│       ├── createdAt: timestamp
│       ├── lastSeen: timestamp
│       └── stats: {
│             totalChallenges: number,
│             bestScore: number,
│             totalPoints: number
│           }
│
├── subjects/                     # المواد
│   └── {subjectId}/
│       ├── id: string
│       ├── name: string
│       ├── nameEn: string
│       ├── icon: string
│       ├── color: string
│       ├── year: number
│       └── isActive: boolean
│
├── questions/                    # الأسئلة
│   └── {questionId}/
│       ├── id: string
│       ├── subjectId: string

│       ├── text: string
│       ├── textEn: string         # (اختياري)

│
├── challenges/                   # سجل التحديات
```

**المبادئ المتبعة:**

**1. Denormalization:**  
في NoSQL، التكرار مقبول لتسريع القراءة.  
مثال: اسم المستخدم في `challenges` بدلاً من جلبه من `users`.

**2. Shallow vs Deep:**  
تجنب Subcollections العميقة جداً (أكثر من 3 مستويات).

**3. Read-optimized:**  
صمّم البيانات بناءً على **كيف ستُقرأ** وليس كيف ستُحفظ.

### **المهمة 1.2: Firebase Security Rules**

**لماذا مهم جداً؟**  
بدون Rules صحيحة، أي شخص يمكنه قراءة أو حذف بيانات قاعدة البيانات!

**الفلسفة:**  
**"Deny by default"** - كل شيء محظور افتراضياً، تسمح فقط بما تحتاجه.

**الـ Rules المطلوبة:**

```javascript
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // === Helper Functions ===
    
    function isAuth() {
      return request.auth != null;
    }
    
    function isOwner(userId) {
      return isAuth() && request.auth.uid == userId;
    }
    
    function hasRole(role) {
      return isAuth() && 
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == role;
    }
    
    function isAdmin() {
      return hasRole('admin');
    }
    
    function isStudent() {
      return hasRole('student') || hasRole('batch_leader') || hasRole('admin');
    }
    
    function isValidQuestion() {
      return request.resource.data.keys().hasAll(['text', 'type', 'subjectId']) &&
             request.resource.data.type in ['mcq', 'essay'];
    }

    // === Users ===
    match /users/{userId} {
      allow read: if isAuth();           // أي مصادق يقرأ
      allow create: if isOwner(userId);  // المستخدم ينشئ ملفه فقط
      allow update: if isOwner(userId) || isAdmin();
      allow delete: if isAdmin();
    }

    // === Questions ===
    match /questions/{questionId} {
      allow read: if isStudent();
      allow create: if isAdmin() && isValidQuestion();
      allow update: if isAdmin();
      allow delete: if isAdmin();
    }

    // === Challenges ===
    match /challenges/{challengeId} {
      allow read: if isOwner(resource.data.userId) || isAdmin();
      allow create: if isOwner(request.resource.data.userId);
      allow update: if false; // لا تعديل على سجلات التحدي
      allow delete: if isAdmin();
    }

    // === Leaderboard ===
    match /leaderboard/{subjectId} {
      allow read: if isAuth();
      allow write: if false; // يُكتب فقط عبر Cloud Functions
    }

    // === Summaries ===
    match /summaries/{summaryId} {
      allow read: if isStudent();
      allow create, update: if isAdmin();
      allow delete: if isAdmin();
    }

    // === Notifications ===
    match /notifications/{userId}/items/{notifId} {
      allow read, update: if isOwner(userId);
      allow create: if isAdmin();
      allow delete: if isOwner(userId);
    }

    // === Settings ===
    match /settings/{document} {
      allow read: if isAuth();
      allow write: if isAdmin();
    }
  }
}
```

**الشرح التفصيلي:**

**`request.auth`:** معلومات المستخدم المصادق (من Firebase Auth)  
**`request.auth.uid`:** الـ ID الفريد للمستخدم  
**`request.resource.data`:** البيانات القادمة (للـ write)  
**`resource.data`:** البيانات الحالية (للـ read)  
**`get()`:** جلب document آخر للتحقق منه


```javascript
/**
 * Global State Manager
 * Simple reactive state for the application
 */

const state = {
  user: null,        // المستخدم الحالي
  isLoading: false,
  language: 'ar',
  theme: 'default',
};

const subscribers = {};

// Get state value
export function getState(key) {
  return state[key];
}

// Update state
export function setState(key, value) {
  state[key] = value;
  notify(key, value);
}

// Subscribe to state changes
export function subscribe(key, callback) {
  if (!subscribers[key]) { subscribers[key] = []; }
  subscribers[key].push(callback);
  
  // Return unsubscribe function
  return () => {
    subscribers[key] = subscribers[key].filter(cb => cb !== callback);
  };
}

// Notify subscribers
function notify(key, value) {
  if (subscribers[key]) {
    subscribers[key].forEach(cb => cb(value));
  }
}
```

```javascript
/**
 * Auth Guard
 * Redirects to login if user is not authenticated
 */
import { getState } from '../core/state.js';

export function requireAuth(callback) {
  const user = getState('user');
  
  if (!user) {
    // Show login modal instead of redirect (better UX)
    openModal('auth-modal');
    return false;
  }
  
  callback(user);
  return true;
}

export function requireRole(role, callback) {
  const user = getState('user');
  
  if (!user) {
    openModal('auth-modal');
    return false;
  }
  
  if (user.role !== role && user.role !== 'admin') {
    Toast.error('غير مصرح لك بهذا الإجراء');
    return false;
  }
  
  callback(user);
  return true;
}
```

```javascript
start() {
  this.startTime = Date.now();
  this.intervalId = setInterval(() => {
    // استخدم Date.now() لقياس الوقت الفعلي
    const elapsed = Math.floor((Date.now() - this.startTime) / 1000);
    const remaining = this.duration - elapsed;
    
    this.onTick(remaining);
    
    if (remaining <= 0) {
      this.stop();
      this.onComplete();
    }
  }, 100); // فحص كل 100ms للدقة
}
```

**لأن Cloud Functions يحتاج Blaze Plan:**

**النهج الثالث (عملي):**
```javascript
async function updateLeaderboard(userId, subjectId, score, time) {
  const leaderRef = doc(db, 'leaderboard', subjectId);
  
  await runTransaction(db, async (transaction) => {
    const leaderDoc = await transaction.get(leaderRef);
    
    let entries = leaderDoc.exists() ? leaderDoc.data().entries : [];
    
    // إضافة النتيجة الجديدة
    entries.push({ userId, name, score, time, date: new Date() });
    
    // ترتيب: أعلى نقطة أولاً، أقل وقت ثانياً
    entries.sort((a, b) => b.score - a.score || a.time - b.time);
    
    // الاحتفاظ بأفضل 10 فقط
    entries = entries.slice(0, 10);
    
    transaction.set(leaderRef, { entries });
  });
}
```

**`runTransaction` ما هو؟**  
عملية atomic: إما تنجح كلها أو تفشل كلها. تمنع race conditions عندما عدة مستخدمين يحدّثون في نفس الوقت.

**الدوال:**

- `getTotalLeaderboard()` - إجمالي جميع المواد

### **المهمة 9.2: Firestore Caching**

**المشكلة:**  
جلب نفس البيانات مرات متعددة = إهدار bandwidth وتكلفة.

**الحل: Memory Cache بسيط**

- [ ] إنشاء `src/js/utils/cache.js`:

```javascript
const cache = new Map();

export function setCache(key, data, ttl = 5 * 60 * 1000) { // 5 minutes
  cache.set(key, {
    data,
    expiresAt: Date.now() + ttl
  });
}

export function getCache(key) {
  const cached = cache.get(key);
  if (!cached) { return null; }
  if (Date.now() > cached.expiresAt) {
    cache.delete(key);
    return null;
  }
  return cached.data;
}

export function clearCache(key) {
  if (key) {
    cache.delete(key);
  } else {
    cache.clear();
  }
}
```

**الاستخدام:**
```javascript
async function getQuestions(subjectId) {
  const cacheKey = `questions_${subjectId}`;
  const cached = getCache(cacheKey);
  
  if (cached) {
    return cached; // بيانات من الـ cache
  }
  
  const questions = await queryDocuments('questions', [
    { field: 'subjectId', op: '==', value: subjectId }
  ]);
  
  setCache(cacheKey, questions);
  return questions;
}
```
