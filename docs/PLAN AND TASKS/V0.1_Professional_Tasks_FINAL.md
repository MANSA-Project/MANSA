### المهمة 4.9: إعداد Firestore Security Rules

**استراتيجية الـ Rules (3 مراحل):**

| المرحلة  | البيئة      | الـ Rules                            | السبب                                      |
| -------- | ----------- | ------------------------------------ | ------------------------------------------ |
| **V0.1** | Development | Firebase Emulator (لا rules تُطبَّق) | Admin panel يكتب بحرية في Dev              |
| **V0.1** | Production  | Read: ✅ كل شيء / Write: ❌ مقفل     | البيانات تُضاف من Firebase Console مباشرةً |
| **V0.5** | Production  | Read: ✅ / Write: حسب الـ role       | بعد إضافة Authentication                   |

**⚠️ المبدأ الأساسي:**

- Admin panel (Phase 16) يعمل **فقط** ضد Firebase Emulator في التطوير
- لإضافة البيانات الأولية للإنتاج → Firebase Console مباشرةً (أو Firebase Admin SDK)
- **لا نفتح Firestore للكتابة العامة في الإنتاج أبداً**

**الخطوات:**

- [ ] تفعيل Firebase Local Emulator:
  ```
  npm install -g firebase-tools
  firebase init emulators
  firebase emulators:start
  ```
- [ ] إضافة `.env.development` متغير: `VITE_USE_EMULATOR=true`
- [ ] في `firebase.js`: connect to emulator إذا `ENV.useEmulator === true`
- [ ] في Firebase Console → Firestore → Rules → نسخ هذه Rules:

```javascript
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ── بيانات المنصة العامة (قراءة للجميع، كتابة للأدمن فقط) ──────────────────
    match /universities/{doc} { allow read: if true; allow write: if false; }
    match /faculties/{doc}    { allow read: if true; allow write: if false; }
    match /departments/{doc}  { allow read: if true; allow write: if false; }
    match /subjects/{doc}     { allow read: if true; allow write: if false; }

    // ── V0.5: سنستبدل write: if false بـ ─────────────────────────────────────
    // allow write: if request.auth != null
    //           && request.auth.token.admin == true;
    //
    // ── V0.5: Users, Challenges, Leaderboard ──────────────────────────────────
    // سيُضاف هنا
  }
}
```

**ملاحظات:**

- Rules الإنتاج تمنع أي كتابة غير مصرح بها
- Emulator في التطوير = لا rules = admin panel يعمل بحرية
- انتقال سلس إلى V0.5: فقط أضف السطرين المعلّقين أعلاه

```javascript
/**
 * MANSA - Global State Management
 * Simple reactive state system
 */

const state = {
  language: 'ar',
  theme: 'default',
  isLoading: false,
  currentRoute: '/',
};

const subscribers = {};

// Get state value
export function getState(key) {
  return state[key];
}

// Set state value + notify subscribers
export function setState(key, value) {
  const oldValue = state[key];
  state[key] = value;

  if (oldValue !== value) {
    notify(key, value);
  }
}

// Subscribe to state changes
export function subscribe(key, callback) {
  if (!subscribers[key]) {
    subscribers[key] = [];
  }
  subscribers[key].push(callback);

  // Return unsubscribe function
  return () => {
    const index = subscribers[key].indexOf(callback);
    if (index > -1) {
      subscribers[key].splice(index, 1);
    }
  };
}

// Notify subscribers
function notify(key, value) {
  if (subscribers[key]) {
    subscribers[key].forEach(callback => callback(value));
  }
}

// Get all state
export function getAllState() {
  return { ...state };
}
```

```javascript
/**
 * MANSA - Hash-based Router
 * Simple SPA routing system
 */

import { setState } from './state.js';
import { Logger } from '../utils/logger.js';

const routes = new Map();
let currentRoute = null;

// Register a route
export function route(path, handler) {
  routes.set(path, handler);
  Logger.debug('Route registered:', path);
}

// Navigate to a route
export function navigate(path, data = {}) {
  window.location.hash = path;
  Logger.info('Navigate to:', path);
}

// Get current route path
export function getCurrentRoute() {
  return window.location.hash.slice(1) || '/';
}

// Extract route params (e.g., /university/:id)
function matchRoute(path) {
  for (const [pattern, handler] of routes) {
    const regex = new RegExp('^' + pattern.replace(/:\w+/g, '([^/]+)') + '$');
    const match = path.match(regex);

    if (match) {
      const params = {};
      const keys = (pattern.match(/:\w+/g) || []).map(k => k.slice(1));
      keys.forEach((key, i) => {
        params[key] = match[i + 1];
      });
      return { handler, params };
    }
  }
  return null;
}

// Handle route change
function handleRouteChange() {
  const path = getCurrentRoute();
  const matched = matchRoute(path);

  if (matched) {
    const { handler, params } = matched;
    currentRoute = path;
    setState('currentRoute', path);
    handler(params);
  } else {
    // 404 - redirect to home
    Logger.warn('Route not found:', path);
    navigate('/');
  }
}

// Initialize router
export function initRouter() {
  // Handle hash change
  window.addEventListener('hashchange', handleRouteChange);

  // Handle initial load
  handleRouteChange();

  Logger.info('Router initialized');
}

// Go back
export function goBack() {
  window.history.back();
}
```

**لماذا هذا ضروري جداً؟**

Firebase Spark (مجاني) = **50,000 قراءة / يوم**

| السيناريو  | قراءات لكل مستخدم        | مع 100 مستخدم      |
| ---------- | ------------------------ | ------------------ |

بدون cache → **تنتهي الحصة المجانية خلال ساعات.**

**ما يُخزَّن في Cache:**

| User data    | ❌ لا cache   | بيانات حساسة ومتغيرة |
| Leaderboard  | 2 دقيقة       | يتغير مع كل تحدي     |

**نوع الـ Cache:** Memory Map مع TTL

```javascript
async function init() {
  Logger.info('Initializing MANSA...');

  try {
    // Register routes
    registerRoutes();

    // Initialize router
    initRouter();

    // Register service worker (PWA)
    if ('serviceWorker' in navigator) {
      window.addEventListener('load', async () => {
        try {
          const registration = await navigator.serviceWorker.register('/service-worker.js');
          Logger.info('Service Worker registered:', registration.scope);
        } catch (error) {
          Logger.error('Service Worker registration failed:', error);
        }
      });
    }

    Logger.info('MANSA initialized successfully');
  } catch (error) {
    Logger.error('Initialization failed:', error);
  }
}

// Start app
init();
```
